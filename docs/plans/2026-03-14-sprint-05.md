```java
// src/main/java/com/example/kreconomonmon/service/GeminiApiService.java
// 기존 클래스에 아래 메서드 추가

/**
 * 컷툰 이미지 생성 요청. 이미지를 지원하지 않는 경우 Optional.empty() 반환.
 */
public Optional<String> generateImage(String prompt) {
    String requestBody = buildImageRequestBody(prompt);
    try {
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(geminiApiUrl + "?key=" + geminiApiKey))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(requestBody))
            .timeout(Duration.ofSeconds(30))
            .build();

        HttpResponse<String> response = httpClient.send(request, HttpResponse.BodyHandlers.ofString());
        if (response.statusCode() != 200) {
            log.warn("Gemini 이미지 생성 실패: HTTP {}", response.statusCode());
            return Optional.empty();
        }
        return extractImageFromResponse(response.body());
    } catch (Exception e) {
        log.warn("Gemini 이미지 생성 중 예외 발생: {}", e.getMessage());
        return Optional.empty();
    }
}

// package-private — 단위 테스트용
String buildImageRequestBody(String prompt) {
    // responseModalities에 IMAGE 포함하여 이미지 생성 요청
    return """
        {
          "contents": [{
            "parts": [{"text": "%s"}]
          }],
          "generationConfig": {
            "responseModalities": ["IMAGE", "TEXT"]
          }
        }
        """.formatted(prompt.replace("\"", "\\\""));
}

// package-private — 단위 테스트용
Optional<String> extractImageFromResponse(String responseBody) {
    try {
        JsonNode root = objectMapper.readTree(responseBody);
        JsonNode parts = root.path("candidates").get(0).path("content").path("parts");
        for (JsonNode part : parts) {
            JsonNode inlineData = part.path("inlineData");
            if (!inlineData.isMissingNode()) {
                String mimeType = inlineData.path("mimeType").asText("image/png");
                String data = inlineData.path("data").asText();
                return Optional.of("data:" + mimeType + ";base64," + data);
            }
        }
    } catch (Exception e) {
        log.warn("Gemini 이미지 응답 파싱 실패: {}", e.getMessage());
    }
    return Optional.empty();
}
```

// src/main/java/com/example/kreconomonmon/service/GeminiAnalysisService.java
// 기존 클래스에 아래 메서드 추가

private static final String ECONOMY_CARTOON_KEY = "ECONOMY_CARTOON";
private static final String REALESTATE_CARTOON_KEY = "REALESTATE_CARTOON";

/**
 * 경제 컷툰 이미지 반환. 캐시 히트 시 Gemini API 미호출.
 * 이미지 생성 불가 시 빈 문자열 반환 (graceful degradation).
 */
public String getEconomyCartoon(String analysisText) {
    return getCartoon(ECONOMY_CARTOON_KEY, analysisText, "한국 경제");
}

/**
 * 부동산 컷툰 이미지 반환. 캐시 히트 시 Gemini API 미호출.
 */
public String getRealEstateCartoon(String analysisText) {
    return getCartoon(REALESTATE_CARTOON_KEY, analysisText, "서울 부동산");
}

private String getCartoon(String cacheKey, String analysisText, String subject) {
    String currentHash = analysisCacheService.computeHash(analysisText);

    Optional<AnalysisCache> cached = analysisCacheService.findByKey(cacheKey);
    if (cached.isPresent() && currentHash.equals(cached.get().getDataHash())) {
        return cached.get().getImageData() != null ? cached.get().getImageData() : "";
    }

    String prompt = buildCartoonPrompt(subject, analysisText);
    Optional<String> imageData = geminiApiService.generateImage(prompt);

    String imageDataValue = imageData.orElse("");
    analysisCacheService.saveOrUpdate(cacheKey, cacheKey + "_TYPE",
        null, imageDataValue.isEmpty() ? null : imageDataValue, currentHash);

    return imageDataValue;
}

private String buildCartoonPrompt(String subject, String analysisText) {
    return String.format(
        "다음 %s 분석 내용을 바탕으로 현재 상황을 표현하는 4컷 만화를 그려주세요. " +
        "귀여운 캐릭터와 간결한 대화로 구성하고, 시각적으로 이해하기 쉽게 표현해주세요. " +
        "분석 내용: %s",
        subject, analysisText
    );
}
```

// 일별(D) 데이터가 1000건 초과 시 샘플링하는 유틸 메서드 추가

/**
 * 데이터 포인트가 maxPoints 초과 시 균등 샘플링 반환.
 * Chart.js 렌더링 성능 보호 목적.
 */
List<EconomyIndicator> sampleIfNeeded(List<EconomyIndicator> data, int maxPoints) {
    if (data.size() <= maxPoints) return data;
    int step = data.size() / maxPoints;
    List<EconomyIndicator> sampled = new ArrayList<>();
    for (int i = 0; i < data.size(); i += step) {
        sampled.add(data.get(i));
    }
    return sampled;
}
```

package com.example.kreconomonmon.controller;

import com.example.kreconomonmon.dto.ApiErrorResponse;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

@Slf4j
@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(Exception.class)
    public ResponseEntity<ApiErrorResponse> handleException(Exception e) {
        log.error("처리되지 않은 예외 발생", e);
        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
            .body(new ApiErrorResponse("데이터를 불러올 수 없습니다", e.getMessage()));
    }

    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<ApiErrorResponse> handleIllegalArgument(IllegalArgumentException e) {
        log.warn("잘못된 요청 파라미터: {}", e.getMessage());
        return ResponseEntity.badRequest()
            .body(new ApiErrorResponse("잘못된 요청입니다", e.getMessage()));
    }
}

4. `browser_console_messages(level: "warning")` -> 경고 최소화

**반응형 최종 검증:**
1. `browser_resize` -> 375x812 (모바일) -> `browser_snapshot`
2. `browser_resize` -> 768x1024 (태블릿) -> `browser_snapshot`
3. `browser_resize` -> 1920x1080 (데스크톱) -> `browser_snapshot`

## 기술 고려사항

### Gemini 이미지 생성 API
- `gemini-2.0-flash` 모델에서 `responseModalities: ["IMAGE"]`를 요청해야 이미지 생성 가능
- 이미지 지원 여부를 런타임에 감지하여 `Optional.empty()` 반환으로 graceful degradation 처리
- Base64 이미지 크기가 클 수 있으므로 `analysis_cache.image_data` 컬럼은 `TEXT` 타입 사용

### 동시 요청 처리
- `ReentrantLock` (per 분석 타입)으로 동시 Gemini 호출 방지
- 탭 전환 시 분석 요청이 빠르게 중복 발생할 수 있으므로 JS 측 debounce도 고려

### 통합 테스트 범위
- `@WebMvcTest`로 컨트롤러 계층 격리 테스트 (DB, 외부 API 불필요)
- 실 DB 연동 테스트는 개발 환경에서 수동 검증으로 대체 (TestContainers 미사용 — YAGNI)

### 배포
- `application-prod.yml`의 `ddl-auto: validate`는 Flyway 마이그레이션(V1~V5) 이후 스키마가 정확해야 함
- 환경변수 미설정 시 앱 구동 실패 방지를 위해 `@Value` 기본값 또는 `@ConditionalOnProperty` 활용 고려
